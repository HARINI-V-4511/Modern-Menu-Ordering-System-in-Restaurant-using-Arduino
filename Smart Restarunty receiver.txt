/*
 * Restaurant Menu System - RECEIVER
 * Hardware: Arduino Uno + RF Receiver + 16x2 LCD (I2C) + Buzzer
 * 
 * Pin Configuration:
 * - RF Receiver Data: Digital Pin 11
 * - RF Receiver VCC: 5V
 * - RF Receiver GND: GND
 * - Buzzer Positive: Digital Pin 2
 * - Buzzer Negative: GND
 * - I2C LCD SDA: A4 (Analog Pin)
 * - I2C LCD SCL: A5 (Analog Pin)
 * - I2C LCD VCC: 5V
 * - I2C LCD GND: GND
 */

#include <RH_ASK.h>
#include <SPI.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// Pin Definitions
const int RF_RX_PIN = 11;    // RF Receiver data pin
const int BUZZER_PIN = 2;    // Buzzer pin
const int LED_PIN = 13;      // Built-in LED for status

// RF Driver - specify RX pin as 11
RH_ASK driver(2000, RF_RX_PIN, 12, 0); // Speed, RX, TX, PTT

// LCD Configuration (I2C address 0x27 or 0x3F, 16 columns, 2 rows)
LiquidCrystal_I2C lcd(0x27, 16, 2); // Try 0x3F if 0x27 doesn't work

// Order Statistics
unsigned long totalOrders = 0;
unsigned long waiterCalls = 0;
unsigned long billRequests = 0;

void setup() {
  Serial.begin(9600);
  
  // Initialize pins
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  
  // Initialize LCD
  lcd.init();
  lcd.backlight();
  lcd.clear();
  
  // Display startup message
  lcd.setCursor(0, 0);
  lcd.print("Restaurant");
  lcd.setCursor(0, 1);
  lcd.print("Order System");
  delay(2000);
  
  // Initialize RF receiver
  Serial.println("=== Restaurant Order Receiver ===");
  Serial.println("Initializing RF receiver on Pin 11...");
  
  if (!driver.init()) {
    Serial.println("RF init failed!");
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("RF Init Failed!");
    while(1); // Stop execution
  } else {
    Serial.println("RF receiver ready!");
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("System Ready!");
    lcd.setCursor(0, 1);
    lcd.print("Waiting...");
    delay(1500);
  }
  
  Serial.println("Waiting for orders...");
  Serial.println("================================");
  
  // Clear LCD for orders
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Waiting for");
  lcd.setCursor(0, 1);
  lcd.print("orders...");
}

void loop() {
  uint8_t buf[RH_ASK_MAX_MESSAGE_LEN];
  uint8_t buflen = sizeof(buf);
  
  if (driver.recv(buf, &buflen)) {
    // Message received!
    digitalWrite(LED_PIN, HIGH);
    
    // Null terminate the received string
    buf[buflen] = '\0';
    String message = String((char*)buf);
    
    // Print to Serial Monitor
    Serial.print("[");
    Serial.print(millis() / 1000);
    Serial.print("s] ORDER RECEIVED: ");
    Serial.println(message);
    
    // Process and display the order
    processOrder(message);
    
    digitalWrite(LED_PIN, LOW);
  }
}

void processOrder(String message) {
  lcd.clear();
  
  if (message.indexOf("Dish1") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 1", totalOrders);
    Serial.println("  → Customer ordered: DISH 1");
  }
  else if (message.indexOf("Dish2") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 2", totalOrders);
    Serial.println("  → Customer ordered: DISH 2");
  }
  else if (message.indexOf("Dish3") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 3", totalOrders);
    Serial.println("  → Customer ordered: DISH 3");
  }
  else if (message.indexOf("Dish4") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 4", totalOrders);
    Serial.println("  → Customer ordered: DISH 4");
  }
  else if (message.indexOf("Dish5") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 5", totalOrders);
    Serial.println("  → Customer ordered: DISH 5");
  }
  else if (message.indexOf("Dish6") >= 0) {
    totalOrders++;
    displayDishOrder("DISH 6", totalOrders);
    Serial.println("  → Customer ordered: DISH 6");
  }
  else if (message.indexOf("CallingWaiter") >= 0) {
    waiterCalls++;
    displayWaiterCall(waiterCalls);
    Serial.println("  → ⚠️  WAITER REQUESTED!");
    alertBuzzer(3); // 3 beeps for waiter call
  }
  else if (message.indexOf("Bill") >= 0) {
    billRequests++;
    displayBillRequest(billRequests);
    Serial.println("  → 💵 BILL REQUESTED");
    alertBuzzer(2); // 2 beeps for bill request
  }
  else {
    // Unknown message
    lcd.setCursor(0, 0);
    lcd.print("Unknown Order");
    lcd.setCursor(0, 1);
    lcd.print(message.substring(0, 16));
    Serial.println("  → Unknown message");
    alertBuzzer(1); // 1 beep for unknown
  }
  
  Serial.println("--------------------------------");
}

void displayDishOrder(String dishName, unsigned long orderNum) {
  // Line 1: Order number
  lcd.setCursor(0, 0);
  lcd.print("Order #");
  lcd.print(orderNum);
  
  // Line 2: Dish name
  lcd.setCursor(0, 1);
  lcd.print(dishName);
  
  // Sound buzzer
  alertBuzzer(1);
}

void displayWaiterCall(unsigned long callNum) {
  // Line 1: Alert message
  lcd.setCursor(0, 0);
  lcd.print("** WAITER **");
  
  // Line 2: Call number
  lcd.setCursor(0, 1);
  lcd.print("Call #");
  lcd.print(callNum);
}

void displayBillRequest(unsigned long requestNum) {
  // Line 1: Bill request
  lcd.setCursor(0, 0);
  lcd.print("BILL REQUEST");
  
  // Line 2: Request number
  lcd.setCursor(0, 1);
  lcd.print("Request #");
  lcd.print(requestNum);
}

void alertBuzzer(int beeps) {
  for (int i = 0; i < beeps; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(150);
    digitalWrite(BUZZER_PIN, LOW);
    delay(150);
  }
}

// Optional: Function to display statistics
void displayStats() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Orders:");
  lcd.print(totalOrders);
  lcd.setCursor(0, 1);
  lcd.print("Calls:");
  lcd.print(waiterCalls);
  lcd.print(" Bills:");
  lcd.print(billRequests);
  delay(3000);
}
